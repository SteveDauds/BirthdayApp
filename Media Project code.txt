using System;
using System.Collections.Generic;
using System.Data;
using System.Data.SQLite;
using System.Linq;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace Phonebook
{
    static class Program
    {
        
        /// <summary>
        /// The main entry point for the application.
        /// </summary>
        [STAThread]
        static void Main()
        {
            string connectionstring;
            connectionstring = @"Data Source =C:\MediaProject\Phonebook.db";
            SQLiteCommand sqlcmd = new SQLiteCommand();
            using (SQLiteConnection con = new SQLiteConnection(connectionstring))
            {
                try
                {
                    con.Open();
                    if (con.State == ConnectionState.Open)
                    {
                        MessageBox.Show("WELCOME!!!");
                    }
                    
                    sqlcmd = con.CreateCommand();
                    sqlcmd.CommandText = "CREATE TABLE tblBirthday (Name TEXT NOT NULL, Birthday varchar(10), PhoneNumber varchar(15));";
                    sqlcmd.ExecuteNonQuery();
                }
                catch (Exception ex)
                {
                    if (MessageBox.Show(ex.Message).ToString().Contains("already"))
                    {
                        MessageBox.Show("Continue...");
                    }
                }
            }
            Application.EnableVisualStyles();
            Application.SetCompatibleTextRenderingDefault(false);
            Application.Run(new Form1());
        }
    }
}

        private void button9_Click(object sender, EventArgs e)
        {
            try
            {
                string Name = txtName.Text.Trim();
                
                using (SQLiteConnection con = new SQLiteConnection(connectionstring))
                {
                    SQLiteCommand cmd = new SQLiteCommand();
                    cmd.CommandText = @"INSERT INTO tblBirthday (Name, Birthday, PhoneNumber) VALUES(+@name, +@birthday, +@phonenumber)";
                    cmd.Connection = con;
                    cmd.Parameters.Add(new SQLiteParameter("@name",Name.ToLower()));                    
                    cmd.Parameters.Add(new SQLiteParameter("@birthday", (txtday.Text+txtmonth.Text).Trim()));
                    cmd.Parameters.Add(new SQLiteParameter("@phonenumber", txtnumber.Text.Trim()));

                    con.Open();

                    int i = cmd.ExecuteNonQuery();

                    if (i == 1)
                    {
                        MessageBox.Show("Birthday Added");
                    }
                    txtName.Clear();
                    txtnumber.Clear();
                    txtday.Text = "DD";
                    txtmonth.Text = "MM";
                }
            }

            {
                MessageBox.Show(ex.Message);
